Module: deps

deps

Module for angularjs style dependency injection.

Members

<inner> contains

Searches to find if a dependency exists.
Example
var deps = require(...);
if (!deps.contains('docker')) {
  deps.register('docker', new Docker(config));
}

<inner> register

Registers dependency for later access by other modules.
Example
var deps = require(...);
deps.register('app', new App(config));

Methods

<static> call(cb)

Calls callback function with auto populated dependencies.
Parameters:
Name Type Description
cb function callback function.
Example
var deps = require(...);
deps.call(function(app, flagVerbose) {
  if (flagVerbose) {
    console.log(app.name);
  }
});

<static> clear()

Clears all registered dependencies.
Example
var deps = require(...);
deps.register('foo', 'bar');
assert(deps.contains('foo'));
deps.clear();
assert(!deps.contains('foo'));

<static> lookup(name, options)

Looks up name of dependency and returns dependency.
Parameters:
Name Type Argument Description
name string Name of dependency.
options object <optional>
 Optional function behaivor.
Returns:
- Value of dependency or undefined if dependency does not exist.
Examples
var deps = require(...);
var result = deps.lookup('mydepname', {optional:true});
if (result) {
  console.log(result);
}
var deps = require(...);
try {
  var result = deps.lookup('mydepname');
} catch (err) {
  console.log('Dependency does not exist!');
}

<static> override(overrides, callback)

Overrides given dependencies with given values for the duration of callback.
Parameters:
Name Type Description
overrides object Object containing key value pairs of overrides.
callback function Callback function to be called to restore overrides.
Example
var deps = require('...');
assert(deps.lookup('mode') === 'prod');
deps.override({mode:test}, function(done) {
  assert(deps.lookup('mode') === 'test');
});

<static> remove(name)

Removes a given dependency.
Parameters:
Name Type Description
name string Name of dependency to remove.
Example
var deps = require(...);
deps.register('key', 'value');
deps.remove('key');
